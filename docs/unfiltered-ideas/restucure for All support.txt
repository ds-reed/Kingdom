restucure for All support

def take(
    self,
    ctx: DispatchContext,
    target: Noun | None,
    words: tuple[str, ...] = (),
) -> str:
    state = self.state(ctx)
    game = self.game(ctx)
    room = self.room(ctx)
    player = self.player(ctx)

    parse = self.resolve_noun_or_word(words, interest=['all'])
    keywords = parse["keywords"]
    is_all = "all" in keywords

    # ───────────────────────────────────────────────
    # Build the list of targets to attempt
    # ───────────────────────────────────────────────
    targets: list[Item] = []

    if is_all:
        # TAKE ALL: collect from floor + open containers + closed containers themselves
        for item in room.items:
            targets.append(item)

        for box in room.boxes:
            if box.is_openable and box.is_open:
                targets.extend(box.contents)
            else:
                targets.append(box)  # try to take the box → likely refuse

        if not targets:
            return self.build_message("There's nothing here to take.")

    else:
        # Single target mode
        if target is None:
            return self.build_message(self.missing_target("take"))

        targets = [target]

    # ───────────────────────────────────────────────
    # Process all targets (1 or many)
    # ───────────────────────────────────────────────
    messages = []
    successes = 0

    for item in targets:
        # 1. Special handler — can STOP everything or SKIP this item
        outcome = self.run_special_handler(item, "take", words, ctx)
        if outcome:
            if outcome.control is VerbControl.STOP:
                # Hard stop: return immediately with this message
                if outcome.message:
                    messages.append(outcome.message)
                break
            if outcome.control is VerbControl.SKIP:
                if outcome.message:
                    messages.append(outcome.message)
                continue

        # 2. Already have it?
        loc = self.locate_item(ctx, item)
        if loc.type == LocationType.INVENTORY:
            messages.append(f"You already have {item.display_name()}.")
            continue

        # 3. Not gettable / special refusal?
        if not getattr(item, "is_takeable", True) or isinstance(item, Box):
            refuse = getattr(item, "get_refuse_string", None) or f"You can't take {item.display_name()}."
            messages.append(refuse)
            continue

        # 4. Perform the take
        success_msg = None
        if loc.type is LocationType.ROOM_FLOOR:
            room.remove_item(item)
            player.add_to_sack(item)
            success_msg = f"You take {item.display_name()}."

        elif loc.type is LocationType.INSIDE_BOX:
            box = loc.container
            box.remove_item(item)
            player.add_to_sack(item)
            success_msg = f"You take {item.display_name()} from {box.display_name()}."

        else:
            messages.append(f"DEBUG: Don't know how to take {item.display_name()} from {loc.type}.")
            continue

        if success_msg:
            messages.append(success_msg)
            successes += 1

    # ───────────────────────────────────────────────
    # Final result
    # ───────────────────────────────────────────────
    if not messages:
        if is_all:
            return self.build_message("There's nothing here to take.")
        else:
            return self.build_message("DEBUG: No message generated for single target.")

    if successes == 0:
        # All failed → just show the refusal(s)
        return self.build_message("\n".join(messages))

    # At least one success
    if is_all:
        # Nice summary for TAKE ALL
        success_lines = [m for m in messages if m.startswith("You take")]
        fail_lines = [m for m in messages if m not in success_lines]

        result = f"You take {successes} item{'s' if successes > 1 else ''}:\n"
        result += "\n".join(f"  • {line}" for line in success_lines)

        if fail_lines:
            result += "\n\nOther items could not be taken:\n"
            result += "\n".join(f"  • {line}" for line in fail_lines)

        return self.build_message(result)

    else:
        # Single item → just return the (usually success) message
        return self.build_message(messages[0])
